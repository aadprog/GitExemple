import requests

# Fonction pour récupérer les données d'une API
def obtenir_donnees_api(url):
    try:
        response = requests.get(url)
        response.raise_for_status()  # Lève une exception en cas d'erreur HTTP

        # Affiche les données de la réponse
        print("Données de l'API:")
        print(response.json())

    except requests.exceptions.HTTPError as errh:
        print(f"Erreur HTTP: {errh}")
    except requests.exceptions.ConnectionError as errc:
        print(f"Erreur de connexion: {errc}")
    except requests.exceptions.Timeout as errt:
        print(f"Timeout: {errt}")
    except requests.exceptions.RequestException as err:
        print(f"Erreur lors de la requête: {err}")

# Exemple d'utilisation avec une API factice
api_url = "https://jsonplaceholder.typicode.com/todos/1"
obtenir_donnees_api(api_url)
