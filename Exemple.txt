class GestionnairePersonnes:
    def __init__(self):
        self.personnes = []

    def ajouter_personne(self, nom, age):
        personne = {"nom": nom, "age": age}
        self.personnes.append(personne)

    def afficher_details(self):
        for personne in self.personnes:
            print(f"Nom: {personne['nom']}, Age: {personne['age']} ans")

    def calculer_moyenne_age(self):
        if not self.personnes:
            return 0
        total_age = sum(personne['age'] for personne in self.personnes)
        moyenne_age = total_age / len(self.personnes)
        return moyenne_age

# Exemple d'utilisation
gestionnaire = GestionnairePersonnes()

gestionnaire.ajouter_personne("Alice", 25)
gestionnaire.ajouter_personne("Bob", 30)
gestionnaire.ajouter_personne("Charlie", 22)

gestionnaire.afficher_details()

moyenne_age = gestionnaire.calculer_moyenne_age()
print(f"Moyenne d'Ã¢ge: {moyenne_age:.2f} ans")
